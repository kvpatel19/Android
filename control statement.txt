fun main(args: Array<String>) { 
println("-----For Loop-------") 
for(item in 1..5) 
{ 
println("$item") 
} 
println("--------While Loop----------") 
var i1 = 0 
while (i1 < 5) { 
println(i1) 
i1++ 
} 
println("----------Do While Loop------------") 
var i: Int = 0 
do{ 
println(i) 
i++; 
}while(i < 3) 
}
